1.1 Zmienne - let

 
        function exampleLet() {
            if (true) {
                let y = 20;
            }
            console.log(y); // Błąd: y is not defined, ponieważ zmienna 'y' jest zdefiniowana tylko wewnątrz bloku 'if'
        }
        
        exampleLet(); // Output: ReferenceError: y is not defined

1.4 Przechowywanie informacji - Array(Tablica)

 
    let numbers = [1, 2, 3, 4, 5];
    console.log(numbers);
    // Output: [1, 2, 3, 4, 5]
    
    console.log(numbers.length); //zwroci informacje o dlugosci naszej tablicy
    // Output: 5
    
    console.log(numbers[0]); // [cyfra] cyfra jest miejscem w tablicy np  miejsce 0 to liczba 1 w naszej tablicy
    // Output: 1
    
    numbers.push(6); // dodaje element 
    console.log(numbers);
    // Output: [1, 2, 3, 4, 5, 6]
    
    numbers.pop(); //usuwa ost element z listy
    console.log(numbers);
    // Output: [1, 2, 3, 4, 5]
    
    console.log(numbers.indexOf(3)); //indexof zwraca pierwsze wystapienie podanego elementu, w tym przypadku jest to 3, i zwrocilo miejsce 2
    // Output: 2

1.4 Przechowywanie informacji - Objekt

 
    let person = { name: "John", age: 30, city: "New York" };
    console.log(person);
    // Output: { name: "John", age: 30, city: "New York" }
    
    console.log(person.name);
    // Output: John
    
    person.country = "USA";
    console.log(person);
    // Output: { name: "John", age: 30, city: "New York", country: "USA" }
    
    delete person.age;
    console.log(person);
    // Output: { name: "John", city: "New York", country: "USA" }
    
    console.log(Object.keys(person));
    // Output: ["name", "city", "country"]
