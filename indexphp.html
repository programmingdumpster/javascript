1.1 Zmienna

 
        // Przykład, gdzie zmienna $y jest dostępna
        function example1() {
            $y = 10;
            echo $y; // Output: 10
        }
        
        example1();
        
        // Przykład, gdzie zmienna $y nie jest dostępna
        function example2() {
            if (false) {
                $y = 20;
            }
            echo $y ?? 'Undefined'; // Output: Undefined
        }
        
        example2();

1.2 Bloki Warunkowe - switch-case

 
    $day = "Monday";

    switch ($day) {
        case "Monday":
            echo "To poniedziałek.";
            break;
        case "Tuesday":
            echo "To wtorek.";
            break;
        case "Wednesday":
            echo "To środa.";

            echo "To inny dzień tygodnia.";
    }

1.3 Bloki Petli - for

1.3 Bloki Petli - while

 
        $i = 0;
        while ($i < 5) {
            echo $i . "\n";
            $i++;
        }
        // Output: 
        // 0
        // 1
        // 2
        // 3
        // 4

1.3 Bloki Petli - do-while

 
        $i = 0;
        do {
            echo $i . "\n";
            $i++;
        } while ($i < 5);
        // Output: 
        // 0
        // 1
        // 2
        // 3
        // 4

1.3 Bloki Petli - foreach

1.4 Przechowywanie informacji - Array(Tablica)

 
    $numbers = [1, 2, 3, 4, 5];
    print_r($numbers);
    // Output: [1, 2, 3, 4, 5]
    
    echo count($numbers); //zwroci informacje o dlugosci naszej tablicy
    // Output: 5
    
    echo $numbers[0]; // [cyfra] cyfra jest miejscem w tablicy np  miejsce 0 to liczba 1 w naszej tablicy
    // Output: 1
    
    $numbers[] = 6; // dodaje element 
    print_r($numbers);
    // Output: [1, 2, 3, 4, 5, 6]
    
    array_pop($numbers); //usuwa ost element z listy
    print_r($numbers);
    // Output: [1, 2, 3, 4, 5]
    
    echo array_search(3, $numbers); //indexof zwraca pierwsze wystapienie podanego elementu, w tym przypadku jest to 3, i zwrocilo miejsce 2
    // Output: 2

1.4 Przechowywanie informacji - Objekt

 
    $person = [ "name" => "John", "age" => 30, "city" => "New York" ];
    print_r($person);
    // Output: { name: "John", age: 30, city: "New York" }
    
    echo $person["name"];
    // Output: John
    
    $person["country"] = "USA";
    print_r($person);
    // Output: { name: "John", age: 30, city: "New York", country: "USA" }
    
    unset($person["age"]);
    print_r($person);
    // Output: { name: "John", city: "New York", country: "USA" }
    
    print_r(array_keys($person));
    // Output: ["name", "city", "country"]
